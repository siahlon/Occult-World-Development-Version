OW_FRXZ_on_actions = {
	first_valid_on_action = {
		OW_FRXZ_on_16th_set_AI #16岁开启信息AI
	}
}
OW_FRXZ_on_16th_set_AI = {
	trigger = { 
		age = 16
		is_ai = yes
	}
	effect = {
		OW_MONTHS_AI_JYXX = yes
	}
}

OW_years_actions = {
	on_actions = {
		OW_XZ_years_actions
	}
	effect = {
		if = {
			limit = {
				is_ai = yes
				age >= 16
				is_alive = yes
				is_imprisoned = no
				NOT = { has_character_flag = OW_xiuzhenzhe }
			}
			OW_MONTHS_AI_JYXX = yes
		}
	}
}
